## Usage

```js
var unixify = require('{%= name %}');

unixify('one\\two\\three');
//=> 'one/two/three'

unixify('one\\two\\//three');
//=> 'one/two/three'

unixify('C:\\one\\two\\three');
//=> '/one/two/three'

unixify('C:\\//one\\//two\\//three');
//=> '/one/two/three'

unixify('C:\\//one\\two\\three');
//=> '/one/two/three'

unixify('C:\\//one\\two\\//three');
//=> '/one/two/three'
```
